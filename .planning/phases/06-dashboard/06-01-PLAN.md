   export function formatBytes(bytes: number, decimals = 1): string {
     if (bytes === 0) return '0 B';
     const k = 1024;
     const sizes = ['B', 'KB', 'MB', 'GB', 'TB'];
     const i = Math.floor(Math.log(bytes) / Math.log(k));
     return parseFloat((bytes / Math.pow(k, i)).toFixed(decimals)) + ' ' + sizes[i];
   }

   export function formatDate(date: string | Date): string {
     return new Intl.DateTimeFormat('en-US', {
       month: 'short', day: 'numeric', year: 'numeric',
       hour: '2-digit', minute: '2-digit',
     }).format(new Date(date));
   }

   export function formatNumber(n: number): string {

|----------|-------------|
| client (sidebar/header) -> server layout | Session data passed from server to client components |
| client -> localStorage | Sidebar collapsed state stored in browser storage |

| Threat ID | Category | Component | Disposition | Mitigation Plan |

| T-06-01 | Spoofing | Dashboard layout | mitigate | Auth guard via auth() + redirect in server layout. Defense-in-depth with middleware proxy. Already established in Phase 2. |
| T-06-02 | Information Disclosure | Header user info | mitigate | Only display email and avatar from session — no sensitive fields. Already pattern from Phase 2 layout. |
| T-06-03 | Tampering | localStorage sidebar state | accept | Sidebar collapsed state is purely cosmetic, no security impact if tampered. |
