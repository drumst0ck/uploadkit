**Step 2: Create `apps/dashboard/src/app/dashboard/layout.tsx`** — Protected layout with session guard.

This is a Server Component that:
1. Calls `auth()` from auth.ts to get the session
2. If no session, `redirect('/login')` — belt-and-suspenders protection beyond proxy.ts (per RESEARCH.md recommendation)
3. If session exists, render the children in a minimal shell

The layout should be minimal in this phase — just a header bar with:
- "UploadKit" brand text (left)
- User avatar (from `session.user.image`) or initials circle (right)
- Sign out button (uses signOut server action)

The sidebar, navigation, and full dashboard chrome come in Phase 6. This phase just needs:
- Proof that the session works
- A sign-out mechanism
- A clean container for the dashboard page content

```typescript
import { auth, signOut } from '@/auth'; // adjust path per tsconfig
import { redirect } from 'next/navigation';
import { connectDB } from '@uploadkit/db';

export default async function DashboardLayout({ children }: { children: React.ReactNode }) {
  // Defense-in-depth: ensure DB connection before auth() to prevent cold-start ordering issues
  await connectDB();
  const session = await auth();
  if (!session?.user) redirect('/login');

  return (
    <div className="min-h-screen bg-[#0a0a0b]">
      <header className="flex items-center justify-between border-b border-white/[0.06] px-6 py-4">
        <span className="text-lg font-semibold text-white">UploadKit</span>
        <div className="flex items-center gap-3">
          {session.user.image && (
            <img src={session.user.image} alt="" className="h-8 w-8 rounded-full" />
          )}
          <span className="text-sm text-zinc-400">{session.user.email}</span>
          <form action={async () => { 'use server'; await signOut({ redirectTo: '/login' }); }}>
            <button type="submit" className="text-sm text-zinc-500 hover:text-zinc-300 transition-colors">
              Sign out
            </button>
          </form>
        </div>
      </header>
      <main className="p-6">{children}</main>
    </div>
  );
}
```

```typescript
import { auth } from '@/auth';
import { redirect } from 'next/navigation';
import { connectDB, Project } from '@uploadkit/db';

export default async function DashboardPage() {
  const session = await auth();
  if (!session?.user) redirect('/login');

  await connectDB();
  const projects = await Project.find({ userId: session.user.id }).lean();

  return (
    <div className="max-w-4xl">
      <h1 className="text-2xl font-semibold text-white mb-2">
        Welcome{session.user.name ? `, ${session.user.name}` : ''}
      </h1>
      <p className="text-zinc-400 mb-8">Here's an overview of your projects.</p>

      {projects.length === 0 ? (
        <div className="rounded-xl border border-white/[0.06] bg-[#141416] p-12 text-center">
          <p className="text-zinc-400 mb-4">No projects yet.</p>
          <p className="text-sm text-zinc-500">Create your first project to get started.</p>
        </div>
      ) : (
        <div className="grid gap-4">
          {projects.map((project) => (
            <div key={String(project._id)} className="rounded-xl border border-white/[0.06] bg-[#141416] p-6">
              <h2 className="text-lg font-medium text-white">{project.name}</h2>
              <p className="text-sm text-zinc-500 mt-1">Slug: {project.slug}</p>
            </div>
          ))}
        </div>
      )}
    </div>
  );
}
```
