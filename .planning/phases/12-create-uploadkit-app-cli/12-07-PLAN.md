      return (

          <main style={{ padding: 48, maxWidth: 640, margin: '0 auto' }}>
            <h1>UploadKit — Vite + React</h1>
            <p>Provide a backend endpoint that returns presigned URLs.</p>
            <UploadDropzone onComplete={(files) => console.log(files)} />
          </main>
        </UploadKitProvider>
      );
    }
