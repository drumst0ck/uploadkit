    export function formatBytes(bytes: number): string {
      if (bytes === 0) return '0 B';
      const k = 1024;
      const sizes = ['B', 'KB', 'MB', 'GB'];
      const i = Math.floor(Math.log(bytes) / Math.log(k));
      return `${parseFloat((bytes / Math.pow(k, i)).toFixed(1))} ${sizes[i]}`;
    }

       ```typescript
       type UploadStatus = 'idle' | 'uploading' | 'success' | 'error';

       type UploadState = {
         status: UploadStatus;
         progress: number;
         error: Error | null;
         result: UploadResult | null;
         abortController: AbortController | null;
       };

       type UploadAction =
         | { type: 'START'; controller: AbortController }
         | { type: 'PROGRESS'; percent: number }
         | { type: 'SUCCESS'; result: UploadResult }
         | { type: 'ERROR'; error: Error }
         | { type: 'RESET' };
       ```

    2. Implement `uploadReducer` function handling all 5 action types:
       - START: status uploading, progress 0, error null, result null, store controller
       - PROGRESS: spread state, update progress
       - SUCCESS: status success, progress 100, store result, null controller
       - ERROR: status error, store error, null controller
       - RESET: return initial state

    3. Export `useUploadKit(route: string)` hook:
       - Calls `useUploadKitContext()` to get client
       - Uses `useReducer(uploadReducer, initialState)`
       - Returns object with:
         - `status`: current UploadStatus
         - `progress`: number (0-100)
         - `error`: Error | null
         - `result`: UploadResult | null
         - `isUploading`: boolean (status === 'uploading')
         - `upload(file: File, metadata?: Record<string, unknown>)`: async function that:
           1. Creates new AbortController
           2. Dispatches START with controller
           3. Calls `client.upload({ file, route, metadata, onProgress: (p) => dispatch({ type: 'PROGRESS', percent: p }), signal: controller.signal })`
           4. On success: dispatches SUCCESS with result
           5. On error: dispatches ERROR with error (unless AbortError — dispatch RESET)
         - `abort()`: calls `state.abortController?.abort()` then dispatches RESET
         - `reset()`: dispatches RESET

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Consumer app -> @uploadkit/react | Untrusted props (apiKey, route, className, appearance) passed to components |
| @uploadkit/react -> @uploadkit/core | SDK client created with consumer's API key; all network calls delegated to core |
| Browser File API -> components | User-selected files processed client-side |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01 | I (Information Disclosure) | UploadKitProvider | mitigate | API key stored in useRef, not in DOM attributes or serialized state; components never render the key |
