```typescript
import { parseModule, generateCode } from "magicast";
const mod = parseModule(source);
mod.imports.$add({ from: "@uploadkitdev/react", imported: "UploadKitProvider" });
const out = generateCode(mod).code;
```

Marker format (D-06):
```
// uploadkit:start
...inserted...
// uploadkit:end
```

Backup manifest shape:
```typescript
interface BackupManifest {
  timestamp: string;             // ISO 8601
  root: string;                  // absolute repo root
  entries: Array<{
    action: "modify" | "create";
    relativePath: string;        // path relative to root
    backupPath?: string;         // only for "modify"
  }>;
}
```
