**Create `apps/api/src/lib/presign.ts`** — presigned PUT URL generator:
```typescript
import { PutObjectCommand } from '@aws-sdk/client-s3';
import { getSignedUrl } from '@aws-sdk/s3-request-presigner';
import { r2Client, R2_BUCKET } from './storage';

// Per D-09: key path = {projectId}/{fileRouterId}/{uuid}/{filename}
export async function generatePresignedPutUrl(params: {
  key: string;
  contentType: string;
  contentLength: number;
  expiresIn?: number;
}): Promise<string> {
  const command = new PutObjectCommand({
    Bucket: R2_BUCKET,
    Key: params.key,
    ContentType: params.contentType,
    ContentLength: params.contentLength,
  });
  return getSignedUrl(r2Client, command, {
    expiresIn: params.expiresIn ?? 900, // 15 minutes
  });
}
```

Note: This enforces ContentType and ContentLength in the signature, preventing type spoofing (Pitfall 1 from PITFALLS.md).

```typescript
import { HeadBucketCommand } from '@aws-sdk/client-s3';
import { r2Client, R2_BUCKET } from '../apps/api/src/lib/storage';

async function verifyR2() {
  console.log('Verifying R2 connection...');
  console.log(`  Bucket: ${R2_BUCKET}`);
  console.log(`  Endpoint: https://${process.env.R2_ACCOUNT_ID}.r2.cloudflarestorage.com`);

  try {
    const result = await r2Client.send(new HeadBucketCommand({ Bucket: R2_BUCKET }));
    console.log('  Status: Connected successfully');
    console.log(`  HTTP Status: ${result.$metadata.httpStatusCode}`);
  } catch (error: any) {
    if (error.name === 'NotFound') {
      console.error(`  ERROR: Bucket '${R2_BUCKET}' does not exist. Create it in Cloudflare dashboard.`);
    } else if (error.name === 'CredentialsProviderError' || error.message?.includes('credentials')) {
      console.error('  ERROR: R2 credentials not configured. Set R2_ACCOUNT_ID, R2_ACCESS_KEY_ID, R2_SECRET_ACCESS_KEY in .env');
    } else {
      console.error(`  ERROR: ${error.message}`);
    }
    process.exit(1);
  }

  console.log('\nR2 verification passed!');
  console.log('\nREMINDER: Configure CORS on the bucket with these settings:');
  console.log(JSON.stringify([{
    AllowedOrigins: ['https://uploadkit.dev', 'https://app.uploadkit.dev', 'http://localhost:3000'],
    AllowedMethods: ['PUT'],
    AllowedHeaders: ['Content-Type', 'Content-Length'],
    MaxAgeSeconds: 3600,
  }], null, 2));

  console.log('\nMANUAL STEPS REQUIRED (see PLAN 01-03 user_setup for full instructions):');
  console.log('  1. Add lifecycle rule in Cloudflare Dashboard -> R2 -> uploadkit-prod -> Lifecycle rules');
  console.log('     Rule: delete objects older than 1 day with UPLOADING status (orphaned presigned uploads)');
  console.log('  2. Connect CDN domain: Cloudflare Dashboard -> R2 -> uploadkit-prod -> Settings -> Custom Domains');
  console.log('     Domain: cdn.uploadkit.dev');

  process.exit(0);
}

verifyR2().catch((err) => {
  console.error('R2 verification failed:', err);
  process.exit(1);
});
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| R2 credentials | Access key + secret stored in env; used server-side only |
| Upstash credentials | REST URL + token stored in env; HTTP-based access |
| Presigned URL | Time-limited, content-type-locked write access to specific R2 key |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-08 | Spoofing | presign.ts | mitigate | ContentType and ContentLength locked in signature; mismatched uploads get 403 from R2 |
| T-01-09 | Tampering | presign.ts | mitigate | Presigned URL expires in 15 minutes (not hours); key path includes projectId for scoping |
| T-01-10 | Denial of Service | rate-limit.ts | mitigate | Sliding window rate limit instances configured (10 req/min general, 30 req/min upload); applied to routes in Phase 3 |
| T-01-11 | Information Disclosure | storage.ts | mitigate | R2 credentials from env vars only; CDN URL separate from S3 API endpoint; CORS locks AllowedOrigins |
| T-01-12 | Information Disclosure | seed.ts | accept | Seed data uses test keys only (`uk_test_`); never contains real credentials; local dev only |
</threat_model>
